422 //--------------------------------------------------------------
423 // Interrupt of Timer2
424 //
425 // CC3-Interrupt -> starts from DMA
426 // Watch it.. higher troughput when interrupt flag is left alone
427 //--------------------------------------------------------------
428 void TIM2_IRQHandler(void)
429 {
430 
431  // Interrupt of Timer2 CH3 occurred (for Trigger start)
432  TIM_ClearITPendingBit(TIM2, TIM_IT_CC3);
433 
434  VGA.hsync_cnt++;
435  if(VGA.hsync_cnt>=VGA_VSYNC_PERIODE)
436  {
437  // -----------
438  VGA.hsync_cnt=0;
439  // Adresspointer first dot
440  VGA.start_adr=(uint32_t)(&VGA_RAM1[0]);
441  }
442 
443  // HSync-Pixel
444  if(VGA.hsync_cnt<VGA_VSYNC_IMP)
445  {
446  // HSync low
447  GPIOB->BSRRH = GPIO_Pin_12;
448  }
449  else {
450  // HSync High
451  GPIOB->BSRRL = GPIO_Pin_12;
452  }
453 
454  // Test for DMA start
455  if((VGA.hsync_cnt>=VGA_VSYNC_BILD_START) && (VGA.hsync_cnt<=VGA_VSYNC_BILD_STOP))
456  {
457  // after FP start => DMA Transfer
458 
459  // DMA2 init
460  DMA2_Stream5->CR=VGA.dma2_cr_reg;
461  // set adress
462  DMA2_Stream5->M0AR=VGA.start_adr;
463  // Timer1 start
464  TIM1->CR1|=TIM_CR1_CEN;
465  // DMA2 enable
466  DMA2_Stream5->CR|=DMA_SxCR_EN;
467 
468  // Test Adrespointer for high
469  if((VGA.hsync_cnt & 0x01)!=0)
470  {
471  // inc after Hsync
472  VGA.start_adr+=(VGA_DISPLAY_X+1);
473  }
474  }
475 }
